class Solution {
    public int findUnsortedSubarray(int[] nums) {
      int i=0;
     int j=0;
     int klast;
     int kfirst;
     int max;
     int min;
     int b=0;
      boolean x=true;
     for(int a=0;a<nums.length-1;a++){
       if(nums[a]>nums[a+1]){
         x=false;
         break;
       }
    }
    for(int k=0;k<nums.length-1;k++) {
       if(nums[k]>nums[k+1]){
         if(i==0 && b==0){
           i=k;
           b=1;
         }
         else{
           j=k+1;
         }
       }
     }
     if(j==0){
       j=i+1;
     }
     int k=i;
     kfirst=i;
     klast=j;
     max=nums[i];
     min=nums[i];
     while(k<=j && k<nums.length){
       if(nums[k]>max){
         max=nums[k];
       }
       if(nums[k]<min){
         min=nums[k];
       }
       k++;
     }
     while(i>=0){
       if(nums[i]>min){
         kfirst=i;
       } 
       i--;
     }
     while(j<nums.length){
       if(nums[j]<max){
         klast=j;
       }
       j++;
     }
     if(x==true){
    return 0;}
    else{
      return (klast-kfirst)+1;
    }
    }
}
